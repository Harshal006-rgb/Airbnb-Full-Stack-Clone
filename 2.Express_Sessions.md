# Express Sessions

An attempt to make our session stateful. Since HTTP is stateless, sessions are used to identify a client across multiple requests.

In **Sessions**, data is stored on the **server-side**, and only a unique session identifier (Session ID) is stored on the **client-side** in a cookie.

---

## 1. Installation

```bash
npm install express-session
```

---

## 2. Setup

```javascript
const session = require("express-session");

const sessionOptions = {
  secret: "mysupersecretstring",
  resave: false,
  saveUninitialized: true,
};

app.use(session(sessionOptions)); // middleware
```

### Options Explained:
* **`secret`**: A string used to sign the session ID cookie (stored on client).
* **`resave`**: Forces the session to be saved back to the session store, even if the session was not modified. (Usually set to `false`).
* **`saveUninitialized`**: Forces a session that is "uninitialized" (new but not modified) to be saved to the store. (Usually set to `true` or `false` depending on use case).

---

## 3. Storing and Accessing Session Data

Session data is stored in `req.session` which is an object. Any variables defined on this object will be persisted across requests for that user.

### Example: Registering Session Variable
```javascript
app.get("/register", (req, res) => {
  let { name = "harshal" } = req.query;
  req.session.name = name;
  res.send(req.session.name);
});
```

---

## 4. Connect Flash

Flash is a special area of the session used for storing messages. Messages are written to the flash and cleared after being displayed to the user (retrieved).

> [!NOTE]
> `connect-flash` requires `express-session` to be set up first.

### 4.1 Installation
```bash
npm install connect-flash
```

### 4.2 Setup
```javascript
const flash = require("connect-flash");

app.use(flash()); // Middleware
```

### 4.3 Storing and Retrieving Messages
Use `req.flash(key, message)` to store, and `req.flash(key)` to retrieve.

```javascript
app.get("/register", (req, res) => {
  let { name = "harshal" } = req.query;
  req.session.name = name;
  req.flash("success", "user registered successfully"); // Store flash msg
  res.redirect("/hello");
});

app.get("/hello", (req, res) => {
  // Retrieve and render flash msg
  res.render("page.ejs", { name: req.session.name, msg: req.flash("success") });
});
```

> [!WARNING]
> **Flash messages appear only once!**
> Once a flash message is read via `req.flash(key)`, it is immediately **removed/cleared** from the session. Any subsequent calls (or page refreshes) will yield an empty array `[]`.

## 5. Complete Classroom Demo Code

Here is the exact code in [classroom/server.js](file:///c:/Users/Lenovo/OneDrive/Documents/Coding/Backend%20Apna%20College/7.Project/classroom/server.js):

```javascript
const express = require("express");
const app = express();
const users = require("./routes/user.js");
const posts = require("./routes/post.js");

const cookieParser = require("cookie-parser");

const session = require("express-session");  //express sessions
const flash = require('connect-flash'); // needs express sessions

const path = require("path");
app.set("view engine", "ejs");        
app.set("views", path.join(__dirname, "views"));

const sessionOptions = {
  secret: "mysupersecretstring" , 
  resave : false, 
  saveUninitialized : true , 
}

app.use(session(sessionOptions)); //middleware
app.use(flash());

app.get("/register" , (req,res)=>{
  let {name = "harshal"} = req.query;
  req.session.name = name;
  req.flash("success", "user registered successfully");
  res.redirect("/hello");
})

app.get("/hello" , (req,res) =>{
  // flash appears only 1 time , after that it becomes empty
  res.render("page.ejs" , {name : req.session.name , msg : req.flash("success")});  
})

// secretcode for signed cookies
app.use(cookieParser("secretcode"));

app.get("/getsignedcookie", (req, res) => {
  res.cookie("color", "red", { signed: true });
  res.send("done!");
});

app.get("/verify", (req, res) => {
  res.send(req.signedCookies);
});

app.get("/getcookies", (req, res) => {
  res.cookie("name", "Harshal");
  res.cookie("madeIn", "India");
  res.send("sent you some cookies!");
});

app.get("/", (req, res) => {
  let { name } = req.cookies;
  res.send(`Hi, ${name}`);
});

app.use("/users", users);
app.use("/posts", posts);

app.listen(3000, () => {
  console.log("server is listening to port 3000");
});
```
